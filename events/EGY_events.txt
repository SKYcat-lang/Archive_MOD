country_event = { # 개혁의 결과
	id = egy.12
	title = egy.12.t
	desc = egy.12.d
	picture = GFX_report_event_EGY_farouk_enthroned

	is_triggered_only = yes

	option = { # 개혁 결과, 다음 단계로...
		name = egy.12.a
		trigger = {
			check_variable = { EGY_reform_stacks > 1 } 
		}
		if = { 
			limit = {  
				AND = {
					check_variable = { EGY_reform_stacks > 1 } 
					check_variable = { EGY_reform_stacks < 15 }
				}
			}
			add_ideas = EGY_conservative_reform
			hidden_effect = {
				country_event = { id = egy.21 days = 1 }
			}
		}
		if = { 
			limit = { check_variable = { EGY_reform_stacks > 14 } }
			add_ideas = EGY_liberal_reform
			hidden_effect = {
				country_event = { id = egy.18 days = 1 }
			}
		}
		
	}
	option = { # 개혁 결과, 다음 단계로...
		name = egy.12.b
		trigger = {
			check_variable = { EGY_reform_stacks < 2 } 
		}
		if = { 
			limit = { check_variable = { EGY_reform_stacks < 2 } }
			add_ideas = EGY_empower_sultan_reform
		}
		hidden_effect = {

		}
	}
