country_event = { #북부 독일 수립
	id = GEN_event.4
	title = GEN_event.4.t
	desc = { 
		text = GEN_event.4.desc
		trigger = {
			has_completed_focus = GEN_communication_with_the_mainland
		}
	}
	desc = { 
		text = GEN_event.4.bad_desc
		trigger = {
			NOT = { has_completed_focus = GEN_communication_with_the_mainland }
		}
	}
	picture = GFX_event_none_picture

	fire_only_once = yes
    is_triggered_only = yes

	trigger = {
		country_exists = GEA
	}

	option = { 
		name = GEN_event.4.a
		every_country = {
			limit = {
				has_war_with = GEA
			}
			ROOT = {
				declare_war_on = {
					target = PREV
					type = annex_everything
				}
			}
		}
		every_state = {
			limit = {
				is_core_of = GEA
			}
			add_core_of = GEN
		}
		GEA = {
			every_character = {
				set_nationality = ROOT
			}
		}
		
		annex_country = {
			target = GEA
			transfer_troops = yes
		}
		complete_national_focus = GEN_get_in_touch
		set_cosmetic_tag = GEN_BAY

		if = {
			limit = {
				NOT = { has_template = "Nationale Miliz" }
			}
			division_template = {
				name = "Nationale Miliz"
				priority = 0
				division_names_group = GER_INF_01
				regiments = {
					militia = { x = 0 y = 0 }
					militia = { x = 0 y = 1 }
					militia = { x = 0 y = 2 }
					
					militia = { x = 1 y = 0 }
					militia = { x = 1 y = 1 }
					militia = { x = 1 y = 2 }
				}
			}
		}
		every_state = {
			limit = {
				has_state_flag = GEN_state_flag
			}
			create_unit = {
				division = "division_template = \"Nationale Miliz\" start_experience_factor = 0.1 start_equipment_factor = 0.6" 
				owner = ROOT
				allow_spawning_on_enemy_provs = yes
			}
			clr_state_flag = GEN_state_flag
		}
	}

	option = { 
		name = GEN_event.4.b
		ai_chance = {
			factor = 0
		}
		add_war_support = -0.20
	}
}
