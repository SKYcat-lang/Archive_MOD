				add_idea = KSP_wakamo_sabotage_2
			}
		}
		if = {
			limit = {
				NOT = {
					check_variable = { KSP_wakamo_next_sabotage = 7 }
				}
			}
			add_to_variable = { KSP_wakamo_next_sabotage = -7 }
		}
	}
