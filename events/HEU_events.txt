		name = HEU_event.8.b

		}

				is_faction_leader = yes

			}

				limit = { is_in_faction_with = GEM }
				GEM = { remove_from_faction = HEU }
			}
		}

			limit = {
				is_in_faction_with = HEU
			}

			}
		}

		}
