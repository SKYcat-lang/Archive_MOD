RED_rejection_of_syndicalism = {

	allowed = { always = yes }
	enable = {
		country_exists = RED
		RED = { has_country_leader_ideology = syndicalism }
		NOT = {
			has_country_leader_ideology = syndicalism
			has_country_leader_ideology = anarchist_communism
		}
		has_government = communism
	}
	abort = {
		OR = {
			is_subject_of = RED
			has_war_with = RED
		}
	}

	ai_strategy = {
		type = alliance
		id = RED
		value = -100
	}
	ai_strategy = {
		type = consider_weak
		id = RED
		value = 50
	}
}
RED_is_syndicalism = {
    reversed = yes
    allowed = {
        original_tag = RED
    }
    enable = {
		NOT = {
			has_country_leader_ideology = syndicalism
			has_country_leader_ideology = anarchist_communism
		}
    }
    enable_reverse = {
		RED = {
			has_country_leader_ideology = syndicalism
		}
    }
    abort_when_not_enabled = yes

    ai_strategy = {
        type = alliance
        id = "RED"
        value = -200
    }
}
