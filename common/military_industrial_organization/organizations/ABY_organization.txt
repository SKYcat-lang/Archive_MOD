ABY_ckd_organization = {
	include = generic_mobile_tank_organization
	icon = GFX_idea_CZE_ckd
	
	allowed = {	
		OR = {
			tag = ABY
			AND = {
				tag = ETH 
				has_dlc = "By Blood Alone"
			}
		}
	}

	visible = {
		IF = {
			limit = { FROM = { original_tag = ETH } }
			FROM = { has_completed_focus = ETH_invite_foreign_industrialists }
		}
	}

	available = {
		# When in a Foreign MIO, countries need to be at peace with original country
		IF = {
			limit = {
				FROM = { NOT = { original_tag = ABY } }
			}
			FROM = { NOT = { has_war_with = CZE } }
		}

		IF = {
			limit = {
				FROM = { original_tag = ETH }
			}
			custom_trigger_tooltip = {
				tooltip = has_invited_mio_tt
				FROM = {
					has_country_flag = has_invited_CZE_ckd_organization_flag
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = ABY_corporate_headquarters_czech_flag
			FROM = {
				NOT = { has_country_flag = ABY_corporate_headquarters_czech_flag } 
			}
		}
	}
}
